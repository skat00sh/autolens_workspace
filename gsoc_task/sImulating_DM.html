1. Visualizing for multiple point masses¶

Best to what I could search, I could not find any utility in PyAutoLens to visualize multiple point masses but it seemed quite possible to just write a basic wrapper around existing functionalities to construct some results.

class MultiPointMass():
    '''
        Class to create an array of dark matter
        
        params:
            centre -> centre of the structure in image plane
            radius -> radius of each point mass in filament. Indirect measure of filament mass
            length -> length of dark matter from centre to end, in arcsecs
            angle -> angle (in degrees) w.r.t the image plane's x-axis
    '''    
        
    def __init__(self, centre = (0, 0), length = 5, radius = 0.5, angle = 0):
        #Initialize with default values
        self.centre = centre
        self.length = length
        self.radius = radius
        self.angle = angle % 360
        
        self.mass_profiles = [al.mp.PointMass(centre = self.centre, einstein_radius = self.radius)]
        
        # determine distance from centre for each point mass
        locations = np.arange(0.25, self.length, 0.25) 
        
        # Calculating coordinates of point masses (Assumtionm: Symmetry around centre)  
        for i in locations:
            x_1 = self.centre[1] + (i * math.cos(math.radians(self.angle))) 
            y_1 = self.centre[0] + (i * math.sin(math.radians(self.angle))) 
            x_2 = self.centre[1] - (i * math.cos(math.radians(self.angle)))
            y_2 = self.centre[0] - (i * math.sin(math.radians(self.angle))) 
             
            
            # Storing all calculated Point Masses in a List
            self.mass_profiles.append(al.mp.PointMass(centre = (x_1, y_1), 
                                                           einstein_radius = self.radius))
            self.mass_profiles.append(al.mp.PointMass(centre = (x_2, y_2), 
                                                           einstein_radius = self.radius))
